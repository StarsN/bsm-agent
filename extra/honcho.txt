from honcho import Honcho

honcho = Honcho(api_key="YOUR_API_KEY")

# Create a session and add a peer
session = honcho.sessions.create(workspace_id="haomeitrade")
honcho.sessions.add_peers("haomeitrade", session.id, {"user1": {"observe_me": True}})

# Add messages
honcho.messages.create("haomeitrade", session.id, "user1", content="I love hiking", is_user=True)

# Query the representation
response = honcho.peers.chat("haomeitrade", "user1", query="What does this user like?")
print(response.content)
